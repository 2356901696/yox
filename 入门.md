## Hello World

```html
<div id="app">
    {{message}}
</div>
```

```javascript
new York({
    el: '#app', 
    data: { 
        message: 'Hello York.js!'
    }
});
```


## 双向绑定

```html
<div id="app">
    <p>
        {{message}}
    </p>
    <input @model="message">
</div>
```

```javascript
new York({
    el: '#app', 
    data: { 
        message: 'Hello York.js!'
    }
});
```

## 渲染列表

```html
<div id="app">
    <ul>
        {{#each todos}}        
        <li>
            {{text}}
        </li>
        {{/each}}
    </ul>
</div>
```


```javascript
new Vue({ 
    el: '#app', 
    data: { 
        todos: [ 
            { text: 'Learn JavaScript' }, 
            { text: 'Learn York.js' }, 
            { text: 'Build Something Awesome' } 
        ]
    }
});
```
